{% macro get_title(lang, original, english) -%}
{% if lang in ('', 'english') -%}
{\large {{ original|texescape -}} } 
{%- if original != english -%} \\ {\itshape {{ english|texescape -}} }{%- endif -%}
{%- elif lang in ('japanese', 'chinese') -%}
\begin{addmargin}[-0.05em]{0em}{\large {{ original|texescape -}} }\end{addmargin}{\itshape {{ english|texescape }} }
{%- elif lang=='korean' -%}
\begin{addmargin}[-0.05em]{0em}{\large{\koreanfont {{ original|texescape -}} } }\end{addmargin}{\itshape {{ english|texescape }} }
{%- else -%}
{\large{\text{{- lang }} {{ original|texescape -}} } } 
{%- if original != english -%} \\ {\itshape {{ english|texescape -}} }{%- endif -%}
{% endif %}
{% endmacro -%}

{% elif display["short_country"] %} ({{ country_abbr|texescape }})
